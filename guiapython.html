Guia Rápido de Python

1. Variáveis e Tipos

# Strings
nome = "Caco"

# Inteiro
idade = 43

# Decimal
altura = 1.75

# Booleano
ativo = True

2. Print e f-strings

print("Olá, mundo!")

nome = "Ana"
idade = 20
print(f"Meu nome é {nome} e tenho {idade} anos")

3. Entrada de dados

nome = input("Digite seu nome: ")
idade = int(input("Digite sua idade: "))
print(f"Olá {nome}, você tem {idade} anos")

4. Condições

idade = 18
if idade >= 18:
    print("Maior de idade")
else:
    print("Menor de idade")

5. Laços de repetição

# For
for i in range(5):
    print(i)

# While
x = 0
while x < 5:
    print(x)
    x += 1

6. Funções

def soma(a, b):
    return a + b

print(soma(3, 4))

7. Estruturas de dados

# Lista
frutas = ["maçã", "banana", "laranja"]

# Dicionário
pessoa = {"nome": "Ana", "idade": 25}

# Tupla
coordenada = (10, 20)

# Conjunto
numeros = {1, 2, 3, 3}
